-- WARNING: This schema is for context only and is not meant to be run.
-- Table order and constraints may not be valid for execution.

CREATE TABLE public.assignment_runs (
  assign_id bigint NOT NULL,
  due_date timestamp with time zone NOT NULL,
  top_k bigint NOT NULL DEFAULT '3'::bigint,
  threshold bigint NOT NULL DEFAULT '70'::bigint,
  assignment_run_id uuid NOT NULL DEFAULT gen_random_uuid(),
  CONSTRAINT assignment_runs_pkey PRIMARY KEY (assignment_run_id),
  CONSTRAINT assignment_runs_assign_id_fkey FOREIGN KEY (assign_id) REFERENCES public.assignments(assign_id)
);
CREATE TABLE public.assignments (
  assign_id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  course_id bigint NOT NULL,
  language text NOT NULL CHECK (language = ANY (ARRAY['java'::text, 'c'::text, 'cpp'::text])),
  name text NOT NULL,
  CONSTRAINT assignments_pkey PRIMARY KEY (assign_id),
  CONSTRAINT Assignments_course_id_fkey FOREIGN KEY (course_id) REFERENCES public.courses(course_id)
);
CREATE TABLE public.courses (
  course_id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  course_name character varying NOT NULL,
  instructor uuid NOT NULL,
  CONSTRAINT courses_pkey PRIMARY KEY (course_id),
  CONSTRAINT courses_instructor_fkey FOREIGN KEY (instructor) REFERENCES auth.users(id)
);
CREATE TABLE public.repositories (
  repository_id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  repository_path text NOT NULL,
  is_default boolean NOT NULL DEFAULT false,
  repository_name text NOT NULL,
  assignment_run_id uuid NOT NULL,
  CONSTRAINT repositories_pkey PRIMARY KEY (repository_id),
  CONSTRAINT repositories_assignment_run_id_fkey FOREIGN KEY (assignment_run_id) REFERENCES public.assignment_runs(assignment_run_id)
);
CREATE TABLE public.results (
  score bigint NOT NULL,
  date_created timestamp with time zone NOT NULL DEFAULT now(),
  submission_1 bigint NOT NULL,
  submission_2 bigint NOT NULL,
  pair_id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  result_path text NOT NULL,
  CONSTRAINT results_pkey PRIMARY KEY (pair_id),
  CONSTRAINT results_submission_1_fkey FOREIGN KEY (submission_1) REFERENCES public.submissions(submission_id),
  CONSTRAINT results_submission_2_fkey FOREIGN KEY (submission_2) REFERENCES public.submissions(submission_id)
);
CREATE TABLE public.submissions (
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  repository_id bigint NOT NULL,
  folder_path text NOT NULL UNIQUE,
  submission_id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  student_first_name_enc text,
  student_last_name_enc text,
  student_email_enc text,
  public_id text,
  CONSTRAINT submissions_pkey PRIMARY KEY (submission_id),
  CONSTRAINT submissions_repository_id_fkey FOREIGN KEY (repository_id) REFERENCES public.repositories(repository_id)
);
